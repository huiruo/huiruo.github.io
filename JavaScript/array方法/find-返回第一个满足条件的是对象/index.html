find只会返回第一个满足条件的是对象​

const testArr = [{ id: 1, name: '张三' }, { id: 2, name: '李四' }, { id: 3, name: '王五' }, { id: 2, name: '赵六' }]

console.log(testArr.find((item) => item.id === 3))
